        obj.put("label", data.getMetadata().getLabel());
        obj.put("groups", data.getString("group"));
        obj.put("path", data.getMetadata().getPath());
        obj.create(actionContext);
        icnt ++;
    }else{
        //println(ds);
        ocnt ++;
    }
    //println(data.getMetadata().getPath());
    task.setCurrentLabel("插入" + icnt + "条记录，已存在" + ocnt + "条记录，总共要处理" + total + "条记录");
    task.setProgress((int) ((icnt + ocnt) * 100 / total));
    //Thread.sleep(1000);
}

task.finished();
