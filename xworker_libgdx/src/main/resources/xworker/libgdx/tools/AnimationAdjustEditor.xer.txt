import org.eclipse.swt.SWT;

if(playCheckButton.isDisposed()){
    return;
}

def gc = event.gc;
def region = null; //要显示的region

//先清屏
def oldColor = gc.getBackground();
gc.setBackground(animationCanvas.getDisplay().getSystemColor(SWT.COLOR_WHITE));
gc.fillRectangle(animationCanvas.getClientArea());
gc.setBackground(oldColor);

//判断是否显示动画
if(playCheckButton.getSelection()){
    def index = actionContext.getScope(0).get("index");
    def lastTime = actionContext.getScope(0).get("lastTime");
   // println("innert:" + index);
    if(index == null){
        index = 0;
    }
    if(lastTime == null){
        lastTime = 0;
    }
    
    def startFrame = Integer.parseInt(startFrameText.getText());
    def endFrame = Integer.parseInt(endFrameText.getText());
    
    //帧速
    def rate = Integer.parseInt(frameRateText.getText());
    
    //当前时间
    def time = System.currentTimeMillis();
    //println("itme-lastTime=" + (time - lastTime));
    if(time - lastTime > rate){
        index++; //播放下一帧
        actionContext.getScope(0).put("lastTime", time);
    }
    if(endFrame >0 && index > endFrame){
        index = 0;
    }
    if(startFrame >0 && index < startFrame){
        index = startFrame;
    }
    if(index < 0 || index >= regions.size()){
        index = 0;
    }
    
    //println(index);
    actionContext.getScope(0).put("index", index);
    
    region = regions.get(index);
    if(region != null){
        region.drawImage(gc);
    }
}else{
    int alpha = gc.getAlpha();
	try{
	    for(item in packTable.getItems()){
		    if(item.getChecked() == true){
		        def id = item.getData().get("id");
		        for(r in regions){
		            if(r.getId() == id){
		                r.drawImage(gc);
		            }
		        }
		    }
		}
		
		gc.setAlpha(128);		
		//当前选中的总是画出来
	    region = actions.doAction("getSelectedRegion", actionContext);
	    if(region != null){
	        region.drawImage(gc);
	    }
	}finally{
		gc.setAlpha(alpha);
	}
}
