def sql = "select ";

//self变量是框架配置，可以获取框架配置的属性和子事物
//self.getChilds("Field")是返回Field子节点的列表
int index = 0;
for(field in self.getChilds("Field")){
    if(index != 0){
        sql = sql + ", ";
    }
    //添加字段的column
    sql = sql + field.getString("column");
    index++;
}

//添加表名
sql = sql + " from " + self.getString("tableName");

//返回生成的SQL
return sql;
