<p>查询数据对象并导出结果。</p>

<p><strong>实现原理</strong></p>

<ol>
	<li>执行的是数据对象的iterator方法。</li>
	<li>必须要设置type属性。</li>
	<li>根据type的值实现三个方法:&lt;type&gt;ExportInit，&lt;type&gt;ExportIter和&lt;type&gt;ExportEnd，比如type=excel，那么就是excelExportInit、excelExportIter和excelExportEnd。</li>
	<li>有一个名为exportContext的变量是Bindings，用于在三个方法中存放自己的变量。</li>
</ol>

<p><strong>输出</strong></p>

<p>&nbsp;&nbsp;&nbsp; 支持OutputStream，File和String，如果是String那么当作文件名。支持var:、ognl:等表达式。</p>

<p><strong>可以重写的方法：</strong></p>

<ul>
	<li><strong>Thing getDataObject()</strong><br />
	返回数据对象的定义事物，默认通过dataObjectPath作为事物路径获取或第一个DataObjects子事物的第一个子事物。<br />
	&nbsp;</li>
	<li><strong>Object getPageInfo()</strong><br />
	返回分页信息，null表示不分页， 默认使用pageInfo属性作为Ognl表达式获取。<br />
	&nbsp;</li>
	<li><strong>Object getConditionData()</strong><br />
	返回条件数据，即查询时用于判断的条件值，默认使用conditionData属性作为Ognl表达式获取。<br />
	&nbsp;</li>
	<li><strong>Thing getConditionConfig()</strong><br />
	返回条件设置，即xworker.dataObject.query.Condition事物，默认使用conditionConfig属性作为Ognl表达式获取，如果没有从第一个Condition子事物获取。<br />
	&nbsp;</li>
	<li><strong>Object getOutput()</strong><br />
	输出，可以是String、File和OutputStream等。默认通过output属性获取，<a href="do?sc=xworker.ide.worldExplorer.swt.http.ThingDoc/@desc&amp;thing=xworker.lang.actions.ActionUtil/@GetData">参看实现文档</a>。<br />
	&nbsp;</li>
	<li><strong>boolean&nbsp;doFilter(DataObject data, Thing dataObject, Bindings exportContext)</strong><br />
	过滤数据，如果返回true，那么该条数据将会被过滤。</li>
</ul>

import xworker.dataObject.DataObject;
import org.xmeta.Thing;
import xworker.task.UserTaskManager;

def userTask = UserTaskManager.createTask(self, false);
userTask.start();
try{
    //用于判断数据是否存在的key
    def fields = self.doAction("getKeyFields", actionContext);
    def keys = fields.split("[,]");
    
    //数据对象和文件
    def dataObject = self.doAction("getDataObject", actionContext);
    def file = self.doAction("getFile", actionContext);
    
    //生成Excel文件
    userTask.setCurrentLabel("生成Excel或Csv数据对象。");
    def fileType = self.doAction("getFileType", actionContext);
    def fileDataObject = null;
   // println("fileType=" + fileType);
    if(fileType == "excel"){
        fileDataObject = new Thing("xworker.dataObject.java.ExcelDataObject");
    }else{
        fileDataObject = new Thing("xworker.dataObject.java.CsvDataObject");
    }
    fileDataObject.put("filePath", file.getAbsolutePath());
    fileDataObject.put("haveHeaders", "true");
    for(attr in dataObject.getChilds("attribute")){
        def excelAttr = null;
        if(fileType == "excel"){
            excelAttr = new Thing("xworker.dataObject.java.ExcelDataObject/@attribute");
        }else{
            excelAttr = new Thing("xworker.dataObject.java.CsvDataObject/@attribute");
        }
        excelAttr.getAttributes().putAll(attr.getAttributes());
        excelAttr.put("propertyPath", attr.metadata.name);
        fileDataObject.addChild(excelAttr);   
    }
    
    //从Excel中读取全部数据
    userTask.setCurrentLabel("开始读取Excel或Csv数据......");
    //println("fileDataObject=" + fileDataObject);
    def datas = fileDataObject.doAction("query", actionContext);
    println("获取数据" + datas.size());
    //改变数据的描述者
    def idatas = [];
    userTask.setCurrentLabel("转换为新的数据对象......");
    for(data in datas){
        def newData = new DataObject(dataObject);
        for(attr in dataObject.getChilds("attribute")){
            newData.put(attr.name, data.get(attr.name));
        }
        idatas.add(newData);    
    }
    
    //创建或者更新数据
    def time = System.currentTimeMillis();
    def count = 0;
    def list = [];
    for(data in idatas){
        list.add(data.createOrUpdate(actionContext, keys));
        //println(data);
        //list.add(data);
        count++;
        if(System.currentTimeMillis() - time > 1000){
            userTask.setCurrentLabel("已合并数据" + count + "条。");
            time = System.currentTimeMillis();
        }
    }
    return list;
}finally{
    userTask.finished();
}
