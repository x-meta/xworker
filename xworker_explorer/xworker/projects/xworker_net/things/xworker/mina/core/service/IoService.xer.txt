import org.apache.mina.core.service.IoServiceListener;

//初始化监听器
for(listeners in self.get("listeners@")){
    for(child in listeners.getChilds()){
        def listener = child.doAction("create", actionContext);
        if(listener instanceof IoServiceListener){
            service.addListener(listener);
        }
    }
}

//初始化过滤器
try{
    def bindings = actionContext.push(null);
    bindings.put("parent", service);
    for(filters in self.get("filters@")){
        for(child in filters.getChilds()){
            child.doAction("create", actionContext);            
        }
    }
}finally{
    actionContext.pop();
}

//设置Handler
def handler = self.get("handler@0");
if(handler != null && handler.getChilds().size() > 0){
    def hd = handler.getChilds().get(0).doAction("create", actionContext);
    if(hd != null){
        service.setHandler(hd);
    }
}
