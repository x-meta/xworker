import org.apache.mina.core.service.IoHandler;

import org.xmeta.ActionContext;

def self = self;
def ac = new ActionContext();
ac.put("parentContext", actionContext);

return [
    exceptionCaught:{
        session,cause->
        self.doAction("exceptionCaught", ac, ["session":session, "cause":cause]);
    },
    messageReceived:{
        session,message->
        self.doAction("messageReceived", ac, ["session":session, "message":message]);
    },
    messageSent:{
        session,message->
        self.doAction("messageSent", ac, ["session":session, "message":message]);
    },
    sessionClosed:{
        session->
        self.doAction("sessionClosed", ac, ["session":session]);
    },
    sessionCreated:{
        session->
        self.doAction("sessionCreated", ac, ["session":session]);
    },
    sessionIdle:{
        session,status->
        self.doAction("sessionIdle", ac, ["session":session, "status":status]);
    },
    sessionOpened:{
        session->
        self.doAction("sessionOpened", ac, ["session":session]);
    },
] as IoHandler;
