#$@text#$@
import org.apache.mina.core.service.IoServiceListener;

import org.xmeta.ActionContext;

def self = self;
def ac = new ActionContext();
ac.put("parentContext", actionContext);

return [
    serviceActivated:{
        service->
        self.doAction("serviceActivated", ac, ["service":service]);
    },
    serviceDeactivated:{
        service->
        self.doAction("serviceDeactivated", ac, ["service":service]);
    },
    serviceIdle:{
        service,idleStatus->
        self.doAction("serviceIdle", ac, ["service":service, "idleStatus":idleStatus]);
    },
    sessionCreated:{
        service->
        self.doAction("sessionCreated", ac, ["service":service]);
    },
    sessionDestroyed:{
        service->
        self.doAction("sessionDestroyed", ac, ["service":service]);
    }
] as IoServiceListener;
#$@text#$@
