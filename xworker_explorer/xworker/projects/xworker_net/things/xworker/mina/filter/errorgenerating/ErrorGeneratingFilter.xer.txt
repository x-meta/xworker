import org.apache.mina.filter.errorgenerating.ErrorGeneratingFilter;

def filter = new ErrorGeneratingFilter();

if(self.changeByteProbability != null && self.changeByteProbability != ""){
    filter.setChangeByteProbability(self.getInt("changeByteProbability"));
}
if(self.duplicatePduProbability != null && self.duplicatePduProbability != ""){
    filter.setDuplicatePduProbability(self.getInt("duplicatePduProbability"));
}
if(self.insertByteProbability != null && self.insertByteProbability != ""){
    filter.setInsertByteProbability(self.getInt("insertByteProbability"));
}
filter.setManipulateReads(self.getBoolean("manipulateReads"));
filter.setManipulateWrites(self.getBoolean("manipulateWrites"));
if(self.maxInsertByte != null && self.maxInsertByte != ""){
    filter.setMaxInsertByte(self.getInt("maxInsertByte"));
}
if(self.removeByteProbability != null && self.removeByteProbability != ""){
    filter.setRemoveByteProbability(self.getInt("removeByteProbability"));
}
if(self.removePduProbability != null && self.removePduProbability != ""){
    filter.setRemovePduProbability(self.getInt("removePduProbability"));
}
if(self.resendPduLasterProbability != null && self.resendPduLasterProbability != ""){
    filter.setResendPduLasterProbability(self.getInt("resendPduLasterProbability"));
}

actionContext.getScope(0).put(self.metadata.name, filter);
if(actionContext.get("parent") != null){
    parent.getFilterChain().addLast(self.metadata.name, filter);
}
return filter;
