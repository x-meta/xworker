#$@text#$@
<p>A filter which intercepts <a title="interface in org.apache.mina.core.service" href="http://mina.apache.org/report/trunk/apidocs/org/apache/mina/core/service/IoHandler.html"><code>IoHandler</code></a> events like Servlet filters. Filters can be used for these purposes:</p>
<ul>
    <li>Event logging,</li>
    <li>Performance measurement,</li>
    <li>Authorization,</li>
    <li>Overload control,</li>
    <li>Message transformation (e.g. encryption and decryption, ...),</li>
    <li>and many more.</li>
</ul>
<p><strong>Please NEVER implement your filters to wrap <a title="interface in org.apache.mina.core.session" href="http://mina.apache.org/report/trunk/apidocs/org/apache/mina/core/session/IoSession.html"><code>IoSession</code></a>s.</strong> Users can cache the reference to the session, which might malfunction if any filters are added or removed later.</p>

import org.apache.mina.core.filterchain.IoFilter;
import org.xmeta.ActionContext;

def self = self;
def ac = new ActionContext();
ac.put("context", actionContext);
def filter = [
    destroy:{
        self.doAction("destroy", ac);
    },
    
    exceptionCaught:{
        nextFilter,session,cause->
        self.doAction("exceptionCaught", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "cause":cause
        ]);
    },
    
    filterClose:{
        nextFilter,session->
        self.doAction("filterClose", ac, [
            "nextFilter":nextFilter,
            "session":session
        ]);
    },
    
    filterSetTrafficMask:{
        nextFilter,session,trafficMask->
        self.doAction("filterSetTrafficMask", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "trafficMask":trafficMask,
        ]);
    },
    
    filterWrite:{
        nextFilter,session,writeRequest->
        self.doAction("filterWrite", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "writeRequest":writeRequest,
        ]);
    },
    
    init:{
        self.doAction("init", ac);
    },
    
    messageReceived:{
        nextFilter,session,message->
        self.doAction("messageReceived", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "message":message,
        ]);
    },
    
    messageSent:{
        nextFilter,session,writeRequest->
        self.doAction("messageSent", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "writeRequest":writeRequest
        ]);
    },
    
    onPreAdd:{
        parent,name,nextFilter->
        self.doAction("onPreAdd", ac, [
            "parent":parent,
            "name":name,
            "nextFilter":nextFilter
        ]);
    },
    
    onPostAdd:{
        parent,name,nextFilter->
        self.doAction("onPostAdd", ac, [
            "parent":parent,
            "name":name,
            "nextFilter":nextFilter
        ]);
    },
    
    onPreRemove:{
        parent,name,nextFilter->
        self.doAction("onPreRemove", ac, [
            "parent":parent,
            "name":name,
            "nextFilter":nextFilter
        ]);
    },
    
    onPostRemove:{
        parent,name,nextFilter->
        self.doAction("onPostRemove", ac, [
            "parent":parent,
            "name":name,
            "nextFilter":nextFilter
        ]);
    },
    
    sessionCreated:{
        nextFilter,session->
        self.doAction("sessionCreated", ac, [
            "nextFilter":nextFilter,
            "session":session
        ]);
    },
    
    sessionOpened:{
        nextFilter,session->
        self.doAction("sessionOpened", ac, [
            "nextFilter":nextFilter,
            "session":session
        ]);
    },
    
    sessionClosed:{
        nextFilter,session->
        self.doAction("sessionClosed", ac, [
            "nextFilter":nextFilter,
            "session":session
        ]);
    },
    
    sessionIdle:{
        nextFilter,session,status->
        self.doAction("sessionIdle", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "status":status
        ]);
    },
    
    exceptionCaught:{
        nextFilter,session,cause->
        self.doAction("exceptionCaught", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "cause":cause
        ]);
    },
    
    messageReceived:{
        nextFilter,session,message->
        self.doAction("messageReceived", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "message":message
        ]);
    },
    
    messageSent:{
        nextFilter,session,message->
        self.doAction("messageSent", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "message":message
        ]);
    },
    
    filterClose:{
        nextFilter,session->
        self.doAction("filterClose", ac, [
            "nextFilter":nextFilter,
            "session":session
        ]);
    },
    
    filterWrite:{
        nextFilter,session,writeRequest->
        self.doAction("filterWrite", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "writeRequest":writeRequest
        ]);
    },
    
    filterSetTrafficMask:{
        nextFilter,session,trafficMask->
        self.doAction("filterSetTrafficMask", ac, [
            "nextFilter":nextFilter,
            "session":session,
            "trafficMask":trafficMask
        ]);
    },
    
    ,
    
    toString:{
        return "mina filter: " + self.metadata.label;
    }
] as IoFilter;

actionContext.getScope(0).put(self.metadata.name, filter);
if(actionContext.get("parent") != null){
    parent.getFilterChain().addLast(self.metadata.name, filter);
}
return filter;
