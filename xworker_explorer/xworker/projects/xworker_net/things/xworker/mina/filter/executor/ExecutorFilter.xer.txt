import org.apache.mina.filter.executor.ExecutorFilter;
import org.apache.mina.core.session.IoEventType;

def executor = null;
def executors = self.get("executor@0");
if(executors != null && executors.getChilds().size() > 0){
    def executorThing = executors.getChilds().get(0);
    executor = executorThing.doAction("create", actionContext);
}

def eventTypes = [];
if(self.getBoolean("CLOSE")){
    eventTypes.add(IoEventType.CLOSE);
}
if(self.getBoolean("EXCEPTION_CAUGHT")){
    eventTypes.add(IoEventType.EXCEPTION_CAUGHT);
}
if(self.getBoolean("MESSAGE_RECEIVED")){
    eventTypes.add(IoEventType.MESSAGE_RECEIVED);
}
if(self.getBoolean("MESSAGE_SENT")){
    eventTypes.add(IoEventType.MESSAGE_SENT);
}
if(self.getBoolean("SESSION_CLOSED")){
    eventTypes.add(IoEventType.SESSION_CLOSED);
}
if(self.getBoolean("SESSION_CREATED")){
    eventTypes.add(IoEventType.SESSION_CREATED);
}
if(self.getBoolean("SESSION_IDLE")){
    eventTypes.add(IoEventType.SESSION_IDLE);
}
if(self.getBoolean("SESSION_OPENED")){
    eventTypes.add(IoEventType.SESSION_OPENED);
}
if(self.getBoolean("WRITE")){
    eventTypes.add(IoEventType.WRITE);
}

def filter = null;
if(executor == null && eventTypes.size() == 0){
    filter = new ExecutorFilter();
}else if(executor != null && eventTypes.size() == 0){
    filter = new ExecutorFilter(executor);
}else if(executor == null && eventTypes.size() > 0){
    filter = new ExecutorFilter(eventTypes as IoEventType[]);
}else{
    filter = new ExecutorFilter(executor, eventTypes as IoEventType[]);
}

actionContext.getScope(0).put(self.metadata.name, filter);
if(actionContext.get("parent") != null){
    parent.getFilterChain().addLast(self.metadata.name, filter);
}
return filter;

<p>A <a title="class or interface in java.util.concurrent" href="http://java.sun.com/j2se/1.5.0/docs/api/java/util/concurrent/ThreadPoolExecutor.html?is-external=true"><code>ThreadPoolExecutor</code></a> that does not maintain the order of <a title="class in org.apache.mina.core.session" href="http://mina.apache.org/report/trunk/apidocs/org/apache/mina/core/session/IoEvent.html"><code>IoEvent</code></a>s. This means more than one event handler methods can be invoked at the same time with mixed order. For xworker.example, let's assume that messageReceived, messageSent, and sessionClosed events are fired.</p>
<ul>
    <li>All event handler methods can be called simultaneously. (e.g. messageReceived and messageSent can be invoked at the same time.)</li>
    <li>The event order can be mixed up. (e.g. sessionClosed or messageSent can be invoked before messageReceived is invoked.)</li>
</ul>
