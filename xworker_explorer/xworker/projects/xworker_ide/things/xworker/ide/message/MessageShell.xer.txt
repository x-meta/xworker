        mainCompositeStackLayout.topControl = thingEditorComposite;
    }else{
        mainCompositeStackLayout.topControl = blankComposite;
    }
}else if(data.type == "composite"){
    //打开Composite
    def thing = world.getThing(data.getString("compositePath"));
    if(thing != null){
        for(child in contentComposite){
            child.dispose();
        }
        
        def ac = new ActionContext();
        ac.put("parent", contentComposite);
        thing.doAction("create", ac);
        contentComposite.layout();
        mainCompositeStackLayout.topControl = contentComposite;
    }else{
        mainCompositeStackLayout.topControl = blankComposite;
    }
}

deleteButton.setEnabled(true);
