import org.xmeta.Thing;
import xworker.ide.tools.autodemo.AutoDemo;

def desc = self.getMetadata().getDescription();
if(desc != null && desc.trim() != "" && desc.trim() != "<p></p>"){
    demo.sysmResponse(self.getMetadata().getPath());
}

//要解说的事物
def thingPath = self.getString("thingPath");
def thing = world.getThing(thingPath);
if(thing == null){
    log.warn("thing is null, demo=" + self.getMetadata.getPath());
    return;
}

//创建演示事物
def demoThing = new Thing("xworker.ide.tools.autodemo.AutoDemo");
//根节点必须田间
createEditThing(demoThing, thing, thing.getString("demo_waitTime"), true);
//初始化子节点
init(demoThing, thing);

//创建子演示
def childDemo = new AutoDemo(demo, demoThing);

def init(demoThing, thing){
    for(child in thing.getChilds()){
        def demo_waitTime = child.getStringBlankAsNull("demo_waitTime");        
        if(demo_waitTime != null){
            createEditThing(demoThing, child, demo_waitTime, false);
        }
        
        init(demoThing, child);
    }
}

def createEditThing(demoThing, thing, waitTime, first){
    //形成说明
    def sys = new Thing("xworker.ide.tools.autodemo.AutoDemo/@SysResponse");
    sys.set("description", thing.getString("description"));
    sys.set("waitTime", "");
    demoThing.addChild(sys);
    
    //打开事物
    def edit = null;
    if(first){
        edit = new Thing("xworker.ide.tools.autodemo.AutoDemo/@EditThing");
        edit.set("isChapter", "true");
    }else{
        edit = new Thing("xworker.ide.tools.autodemo.AutoDemo/@ThingEditorSelectNode");
    }
    edit.set("thingPath", thing.getMetadata().getPath());
    edit.set("waitTime", waitTime);
    demoThing.addChild(edit);
    
    
}
