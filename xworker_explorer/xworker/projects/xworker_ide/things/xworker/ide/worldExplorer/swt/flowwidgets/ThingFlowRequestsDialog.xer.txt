import org.eclipse.swt.SWT;
import org.eclipse.swt.widgets.MessageBox;

if(event.widget == cancelButton){
    shell.dispose();
    return;
}else if(event.widget == openButton){
    def request = requestsTable.getSelection()[0].getData();
    
    //查看当前正在执行的请求是否已经完成           
    if(parentContext.get("currentRequestFinished") == false){
        MessageBox box = new MessageBox(openButton.getShell(), SWT.YES | SWT.NO | SWT.ICON_QUESTION);
        box.setText("操作信息");
        box.setMessage("原先的请求还未处理完毕，要放弃原先的请求而处理新的请求吗？");
        if(box.open() == SWT.NO){
            return;
        }
    }

    //设置编辑请求事物，并选择要编辑的节点
    def scripts = parentContext.actions;        
    if(request.thing.getRoot() != parentContext.get("currentThing").getRoot()){  
        scripts.doAciton("setThing", ["thing": request.thing.getRoot()]);                     
    }
    scripts.doAction("selectThing", ["thing":request.thing]);        
    
    shell.dispose();
}
