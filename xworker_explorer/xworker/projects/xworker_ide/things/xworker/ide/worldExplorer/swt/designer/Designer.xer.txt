import groovy.lang.Binding;

import org.eclipse.swt.SWT;
import org.xmeta.ActionContext;

import xworker.ui.swt.SwtBorder;
import xworker.swt.design.Designer;

dataObject = world.getThing(dataObject.metadata.path);

for(child in contentComposite.getChildren()){
    child.dispose();
    contentComposite.removeControl(child);
}

def oldComposite = contentComposite.getContent();
if(oldComposite != null){
    oldComposite.dispose();
}

ActionContext ac = new ActionContext();
dataObject = dataObject.detach();
dataObject.set("descriptors", "xworker.swt.widgets.Composite");
ac.put("parent", contentComposite);
def newComposite = dataObject.doAction("create", ac);
SwtBorder.attach(newComposite);
contentComposite.setContent(newComposite);
newComposite.setSize(newComposite.computeSize(SWT.DEFAULT, SWT.DEFAULT));

Designer.attach(newComposite, dataObject.metadata.path, ac);
