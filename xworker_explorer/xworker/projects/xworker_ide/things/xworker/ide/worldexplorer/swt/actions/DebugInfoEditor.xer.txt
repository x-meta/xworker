import org.eclipse.swt.widgets.TreeItem;

import java.text.SimpleDateFormat;

def recordsTable = actionRecordsTable;
def store = store;
def log = log;
actionRecordsTable.display.asyncExec({
    try{
        for(item in recordsTable.items){
            item.dispose();
        }
        
        def items = [:];
        
        SimpleDateFormat sf = new SimpleDateFormat("HH:mm:ss.SSS");
        for(record in store.records){
            def parentItem = null;
            for(int i=record.scopeSize - 1; i>=0; i--){
                parentItem = items.get(record.thread + "-" + i);
                if(parentItem != null){
                    break;
                }
            }
            if(parentItem == null){
                parentItem = recordsTable;
            }
            
            def item = new TreeItem(parentItem, SWT.NONE);
            //def label = record.method;
            def texts = [record.scopeSize + " " + record.name, sf.format(record.time), record.thread,  record.type, record.path] as String[];
            item.setText(texts);
            item.setData(record);
            items.put(record.thread + "-" + record.scopeSize, item);
            def sourceData = record.get("__ListDataObject_orginal_data__");
            record.parameters = sourceData.parameters;
        }
    }catch(Exception e){
        log.error("init action tree error", e);
    }

    //启动动作记录
    if(Debuger.getDebuger().isEnabled() == false){
        def box = new MessageBox(actionRecordButton.getShell(), SWT.ICON_WARNING | SWT.OK);
        box.setText(UtilString.getString("lang:d=录制动作&en=Record Action", actionContext));
        box.setMessage(UtilString.getString("lang:d=需要先打开调试。&en=Need enable debug first.", actionContext));

    Debuger.getDebuger().startActionRecord();
    actionRecordButton.setText(UtilString.getString("lang:d=停止录制&en=Stop Record", actionContext));
    actionRecordButton.setData("started", true);

    Debuger.getDebuger().stopActionRecord();

    actionRecords.clear();
    actionRecordThreads.clear();

    //println(actionRecords);

    actionRecordButton.setData("started", false);
    
    acRecordStore.doAction("load", actionContext);
    acRecordThreadStore.doAction("load", actionContext);

def data = event.item.getData(); //动作记录
def thing = world.getThing(data.path);
if(thing != null){
    //设置事物
    actionRecordForm.doAction("setThing", actionContext, ["thing": thing]);
    
    //设置参数列表
    for(item in acrParamsTable.getItems()){
        item.dispose();
    }
    if(data.parameters != null){
        def list = [];
        for(key in data.parameters.keySet()){
            list.add(key);
        }
        java.util.Collections.sort(list);
        
        for(name in list){
            def item = new TableItem(acrParamsTable, SWT.NONE);
            item.setData(name);
            item.setText([name] as String[]);
        }
    }
    
    //设置路径标签
    acReocrdTingPathLabel.setText(thing.getMetadata().getPath());
    acReocrdTingPathLabel.update();
}
