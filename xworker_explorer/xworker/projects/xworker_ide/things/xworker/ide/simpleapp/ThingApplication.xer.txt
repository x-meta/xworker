if(thing.getActionThing("run") == null){
    def box = new MessageBox(shell, SWT.ICON_WARNING | SWT.OK);
    box.setText("执行");
    box.setMessage("正在编辑的事物没有run方法！");
    box.open();
    return;
}

try{
    thing.doAction("run", actionContext);
    def box = new MessageBox(shell, SWT.ICON_WARNING | SWT.OK);
    box.setText("执行");
    box.setMessage("成功执行完毕！");
    box.open();
}catch(Exception e){
    e.printStackTrace();
    def box = new MessageBox(shell, SWT.ICON_WARNING | SWT.OK);
    box.setText("执行");
    box.setMessage("执行错误，" + ExceptionUtil.getRootMessage(e));
    box.open();
}
