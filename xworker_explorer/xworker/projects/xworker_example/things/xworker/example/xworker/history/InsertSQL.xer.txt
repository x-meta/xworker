def sql = "insert into ${self.tableName}(";
def valuesSql ="";
def fields = self.getChilds("Field");
for(int i=0; i<fields.size(); i++){
    def child = fields.get(i);
    if(i != 0){
        sql = sql + ",";
        valuesSql = valuesSql + ",";
    }
    sql = sql + child.name;
    valuesSql = valuesSql + "?";
}

return sql + ") values(" + valuesSql + ")";
