import org.eclipse.draw2d.Orientable;

import org.eclipse.draw2d.ScrollBar;

def figure = new ScrollBar();
switch(self.direction){
    case "HORIZONTAL":
        figure.setDirection(Orientable.HORIZONTAL);
        break;
    case "VERTICAL":
        figure.setDirection(Orientable.VERTICAL);
        break;
}
if(self.extent != null && self.extent != ""){
    figure.setExtent(self.getInt("extent"));
}
if(self.maximum != null && self.maximum != ""){
    figure.setMaximum(self.getInt("maximum"));
}
if(self.minimum != null && self.minimum != ""){
    figure.setMinimum(self.getInt("minimum"));
}
if(self.pageIncrement != null && self.pageIncrement != ""){
    figure.setPageIncrement(self.getInt("pageIncrement"));
}
if(self.stepIncrement != null && self.stepIncrement != ""){
    figure.setStepIncrement(self.getInt("stepIncrement"));
}
if(self.value != null && self.value != ""){
    figure.setValue(self.getInt("value"));
}

return figure;
