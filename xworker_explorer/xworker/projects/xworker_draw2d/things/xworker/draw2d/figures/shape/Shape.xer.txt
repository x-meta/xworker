import org.xmeta.util.UtilString;

import org.eclipse.swt.SWT;


if(self.alpha != null && self.alpha!= ""){
    figure.setAlpha(self.getInt("alpha"));
}

if(self.antialias != null && self.antialias!= ""){
    figure.setAntialias(self.getInt("antialias"));
}

figure.setFill(self.getBoolean("fill"));
figure.setFillXOR(self.getBoolean("fillXOR"));

if(self.lineAttributes != null && self.lineAttributes != ""){
    def lineAttributes = UtilString.createObjectFromParams(self.lineAttributes, "xworker.swt.graphics.LineAttributes", actionContext);
    if(lineAttributes != null){
        figure.setLineAttributes(lineAttributes);
    }
}

if(self.lineCap != null && self.lineCap!= ""){
    def cap = 0;
    switch(self.lineCap){
        case "CAP_FLAT":
            figure.setLineCap(SWT.CAP_FLAT);
            break;
        case "CAP_ROUND":
            figure.setLineCap(SWT.CAP_ROUND);
            break;
        case "CAP_SQUARE":
            figure.setLineCap(SWT.CAP_SQUARE);
            break;
    }
}

if(self.lineDash != null && self.lineDash!= ""){
    def dash = null;
    if(self.dash != null && self.dash != ""){
        def dashs = [];
        for(dstr in self.dash.split("[,]")){
            dashs.add(Float.parseFloat(dstr));
        }
        dash = dashs as Float[];
    }
    figure.setLineDash(dash);
}

if(self.lineDashOffset != null && self.lineDashOffset!= ""){
    figure.setLineDashOffset(self.getFloat("lineDashOffset"));
}

if(self.lineJoin != null && self.lineJoin!= ""){
    switch(self.join){
        case "JOIN_BEVEL":
            figure.setLineJoin(SWT.JOIN_BEVEL);
            break;
        case "JOIN_MITER":
            figure.setLineJoin(SWT.JOIN_MITER);
            break;
        case "JOIN_ROUND":
            figure.setLineJoin(SWT.JOIN_ROUND);
            break;
    }
}

if(self.lineMiterLimit != null && self.lineMiterLimit!= ""){
    figure.setLineMiterLimit(self.getFloat("lineMiterLimit"));
}

if(self.lineStyle != null && self.lineStyle!= ""){
    switch(self.style){
        case "LINE_CUSTOM":
            figure.setLineStyle(SWT.LINE_CUSTOM);
            break;
        case "LINE_DASH":
            figure.setLineStyle(SWT.LINE_DASH);
            break;
        case "LINE_DASHDOT":
            figure.setLineStyle(SWT.LINE_DASHDOT);
            break;
        case "LINE_DASHDOTDOT":
            figure.setLineStyle(SWT.LINE_DASHDOTDOT);
            break;
        case "LINE_DOT":
            figure.setLineStyle(SWT.LINE_DOT);
            break;
        case "LINE_SOLID":
            figure.setLineStyle(SWT.LINE_SOLID);
            break;
    }
}

if(self.lineWidth != null && self.lineWidth != ""){
    figure.setLineWidth(self.getInt("lineWidth"));
}

if(self.lineWidthFloat != null && self.lineWidthFloat != ""){
    figure.setLineWidthFloat(self.getFloat("lineWidthFloat"));
}

figure.setOutline(self.getBoolean("outline"));
figure.setOutlineXOR(self.getBoolean("outlineXOR"));
figure.setXOR(self.getBoolean("XOR"));
