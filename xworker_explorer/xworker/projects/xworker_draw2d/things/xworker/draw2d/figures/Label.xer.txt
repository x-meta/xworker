import org.eclipse.draw2d.Label;
import org.eclipse.draw2d.PositionConstants;

import org.xmeta.util.UtilString;

def figure = new Label();
def Constants = world.getActionClass("xworker.draw2d.constants.ConstantsAction", actionContext);

def image = UtilString.createObjectFromParams(self.icon, "xworker.swt.graphics.Image", actionContext);
if(image != null){
    figure.setIcon(image);
}
if(self.iconAlignment != null && self.iconAlignment != ""){
    figure.setIconAlignment(Constants.getPositionConstants(self.iconAlignment));
}
figure.setIconTextGap(self.getInt("iconTextGap"));
if(self.labelAlignment != null && self.labelAlignment != ""){
    figure.setLabelAlignment(Constants.getPositionConstants(self.labelAlignment));
}
if(self.textAlignment != null && self.textAlignment != ""){
    figure.setTextAlignment(Constants.getPositionConstants(self.textAlignment));
}
if(self.textPlacement != null && self.textPlacement != ""){
    figure.setTextPlacement(Constants.getPositionConstants(self.textPlacement));
}
if(self.text != null && self.text != ""){
    figure.setText(self.text);
}
return figure;
