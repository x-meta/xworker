import ognl.Ognl;

import org.eclipse.draw2d.geometry.PointList;

if(self.type == "Var"){
    def points = Ognl.getValue(self.varName, actionContext);
    if(points != null){
        figure.setTemplate(points);
    }
    
    return null;
}

PointList list = new PointList();
for(child in self.getChilds()){
    def point = child.doAction("create", actionContext);
    if(point != null){
        list.addPoint(point);
    }
}
figure.setTemplate(list);

return null;
