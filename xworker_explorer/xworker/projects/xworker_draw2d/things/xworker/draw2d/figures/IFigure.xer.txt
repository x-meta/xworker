import org.eclipse.draw2d.IFigure;

//log.info("create figure");
//创建Figure 
def figure = self.doAction("createFigure", actionContext);
if(figure == null){
    return null;
}

try{
    def bindings = actionContext.push(null);
    bindings.put("figure", figure);
    //bindings.put("parent", figure);
    
    //创建子节点
    for(child in self.getChilds()){
        def obj = child.doAction("create", actionContext);        
        if(obj instanceof IFigure){
            log.info("obj=" + obj + ",childPath=" + child.metadata.path);
            figure.add(obj);
        }
    }
    
    //初始化属性等
    if(figure != null){
        self.doAction("init", actionContext);
    }
}finally{
    actionContext.pop();
}

//保存变量
//log.info("name=" + self.metadata.name + ",actionContextVar=" + self.actionContextVar);
if(self.getBoolean("actionContextVar")){
    actionContext.getScope(0).put(self.metadata.name, figure);
}

return figure;

    if(childFigure != null){
