#$@text#$@
<p>A <code>FlowFigure</code> represented by a single <a title="class in org.eclipse.draw2d.text" href="http://help.eclipse.org/galileo/topic/org.eclipse.draw2d.doc.isv/reference/api/org/eclipse/draw2d/text/BlockBox.html"><code>BlockBox</code></a> containing one or more lines. A BlockFlow is a creator of LineBoxes, which its children require during layout. A BlockFlow can be thought of as a foundation for a paragraph.</p>
<p>BlockFlows must be parented by a <code>FlowFigure</code>. <a title="class in org.eclipse.draw2d.text" href="http://help.eclipse.org/galileo/topic/org.eclipse.draw2d.doc.isv/reference/api/org/eclipse/draw2d/text/FlowPage.html"><code>FlowPage</code></a> can be used as a &quot;root&quot; block and can be parented by normal Figures.</p>
<p>Only <a title="class in org.eclipse.draw2d.text" href="http://help.eclipse.org/galileo/topic/org.eclipse.draw2d.doc.isv/reference/api/org/eclipse/draw2d/text/FlowFigure.html"><code>FlowFigure</code></a>s can be added to a BlockFlow.</p>

import org.eclipse.draw2d.PositionConstants;

import org.eclipse.swt.SWT;

if(self.horizontalAligment != null && self.horizontalAligment != ""){
    switch(self.horizontalAligment){
        case "LEFT":
            figure.setHorizontalAligment(PositionConstants.LEFT);
            break;
        case "RIGHT":
            figure.setHorizontalAligment(PositionConstants.RIGHT);
            break;
        case "CENTER":
            figure.setHorizontalAligment(PositionConstants.CENTER);
            break;
        case "ALWAYS_LEFT":
            figure.setHorizontalAligment(PositionConstants.ALWAYS_LEFT);
            break;
        case "ALWAYS_RIGHT":
            figure.setHorizontalAligment(PositionConstants.ALWAYS_RIGHT);
            break;
    }
}

if(self.orientation != null && self.orientation != ""){
    switch(self.orientation){
        case "LEFT_TO_RIGHT":
            figure.setOrientation(SWT.LEFT_TO_RIGHT);
            break;
        case "RIGHT_TO_LEFT":
            figure.setOrientation(SWT.RIGHT_TO_LEFT);
            break;
    }
}
