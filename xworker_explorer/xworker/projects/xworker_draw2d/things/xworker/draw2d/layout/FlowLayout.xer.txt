import org.eclipse.draw2d.FlowLayout;

def layout = new FlowLayout();

layout.setObserveVisibility(self.getBoolean("observeVisibility"));

layout.setHorizontal(self.getBoolean("isHorizontal"));
layout.setStretchMinorAxis(self.getBoolean("stretchMinorAxis"));

switch(self.majorAlignment){
    case "ALIGN_CENTER":
        layout.setMajorAlignment(FlowLayout.ALIGN_CENTER);
        break;
    case "ALIGN_LEFTTOP":
        layout.setMajorAlignment(FlowLayout.ALIGN_LEFTTOP);
        break;
    case "ALIGN_RIGHTBOTTOM":
        layout.setMajorAlignment(FlowLayout.ALIGN_RIGHTBOTTOM);
        break;
}
if(self.majorSpacing != null && self.majorSpacing != ""){
    layout.setMajorSpacing(self.getInt("majorSpacing"));
}

switch(self.minorAlignment){
    case "ALIGN_CENTER":
        layout.setMinorAlignment(FlowLayout.ALIGN_CENTER);
        break;
    case "ALIGN_LEFTTOP":
        layout.setMinorAlignment(FlowLayout.ALIGN_LEFTTOP);
        break;
    case "ALIGN_RIGHTBOTTOM":
        layout.setMinorAlignment(FlowLayout.ALIGN_RIGHTBOTTOM);
        break;
}
if(self.minorSpacing != null && self.minorSpacing != ""){
    layout.setMinorSpacing(self.getInt("minorSpacing"));
}

return layout;
