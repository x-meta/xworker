import org.eclipse.draw2d.DefaultRangeModel;

import ognl.Ognl;

if(self.type == "Var"){
    return Ognl.getValue(self.varName);
}

DefaultRangeModel model = new DefaultRangeModel();
model.setAll(self.getInt("minimum"), self.getInt("extent"), self.getInt("maximum"));
if(self.value != null && self.value != ""){
    return model.setValue(self.getInt("value"));
}

return model;
