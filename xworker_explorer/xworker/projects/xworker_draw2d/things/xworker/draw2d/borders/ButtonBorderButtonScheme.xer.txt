import org.eclipse.draw2d.ButtonBorder.ButtonScheme;

import org.eclipse.swt.graphics.Color;

switch(self.type){
    case "topbottompresse":
        def topleft = createColors(self, "Topleft", actionContext);
        def bottomright = createColors(self, "Bottomright", actionContext);
        def topleftPressed = createColors(self, "TopleftPressed", actionContext);
        def bottomrightPressed = createColors(self, "BottomrightPressed", actionContext);
        return new ButtonScheme(topleft,bottomright,topleftPressed,bottomrightPressed);
    case "highlightShadow":
        def highlights = createColors(self, "Highlights", actionContext);
        def shadows = createColors(self, "Shadows", actionContext);
        return new ButtonScheme(highlights, shadows);
}
return null;

def createColors(thing, name, actionContext){
    def colors = [];
    def colorsThing = thing.get(name + "@0");
    if(colorsThing != null){
        for(colorThing in colorsThing.getChilds()){
            def color = colorThing.doAction("create", actionContext);
            if(color != null){
                colors.add(color);
            }
        }
    }
    
    return colors as Color[];
}
