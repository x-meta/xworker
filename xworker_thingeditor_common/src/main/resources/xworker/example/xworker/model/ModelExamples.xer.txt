import org.eclipse.swt.SWT;
import org.eclipse.swt.widgets.Display;
import org.eclipse.swt.widgets.Shell;
import org.eclipse.swt.widgets.Text;

public class SWTHelloWorld {
    public static void main (String [] args) {
	    Display display = new Display ();
    	Shell shell = new Shell(display);
	
    	Text helloWorldTest = new Text(shell, SWT.NONE);
    	helloWorldTest.setText("Hello World SWT");
    	helloWorldTest.pack();
	
    	shell.pack();
    	shell.open ();
    	while (!shell.isDisposed ()) {
    		if (!display.readAndDispatch ()) display.sleep ();
    	}
    	display.dispose ();
    }
}

Ext.onReady(function() {
    Ext.QuickTips.init();

    Ext.BLANK_IMAGE_URL='js/extjs/resources/images/default/s.gif';

    var provinceDataStore = new Ext.data.JsonStore({
        idProperty: 'id',
        root: 'rows',
        successProperty: 'success',
        totalProperty: 'totalCount',
        messageProperty: 'msg',
        storeId: 'provinceDataStore',
        autoLoad: true,
        proxy: new Ext.data.HttpProxy({
            timeout: 180000,
            api: {
                read: 'do?sc=xworker.app.view.extjs.server.DataObjectStore/@read',
                create: 'do?sc=xworker.app.view.extjs.server.DataObjectStore/@create',
                update: 'do?sc=xworker.app.view.extjs.server.DataObjectStore/@update',
                destroy: 'do?sc=xworker.app.view.extjs.server.DataObjectStore/@destroy'
            }
        }),
        writer: new Ext.data.JsonWriter({
            encodeDelete: true
        }),
        fields: [
            {
                mapping: 'id',
                name: 'id',
                type: 'int'
            },
            {
                mapping: 'value',
                name: 'value'
            }
        ],
        baseParams: {
            dataObjectPath:'xworker.example.app.dataobject.ProvinceDataObject'
        },
        listeners: {
            exception: function(proxy, type, action, options, response, arg){
                
                var msg = '';
                if(Ext.isDefined(response.responseText)){
                    var o = Ext.util.JSON.decode(response.responseText);
                    msg = o.msg;
                }else{
                    msg = response.message;
                }
                if(msg){
                    Ext.Msg.alert(type, msg);
                }else{
                    Ext.Msg.alert(type, '超时或发生其他错误,responseText=' + response.responseText);
                }
            }
        }
    });

    var gridPanel_store = new Ext.data.JsonStore({
        idProperty: 'id',
        root: 'rows',
        successProperty: 'success',
        totalProperty: 'totalCount',
        messageProperty: 'msg',
        storeId: 'gridPanel_store',
        autoLoad: true,
        proxy: new Ext.data.HttpProxy({
            timeout: 180000,
            api: {
                read: 'do?sc=xworker.app.view.extjs.server.DataObjectStore/@read',
                create: 'do?sc=xworker.app.view.extjs.server.DataObjectStore/@create',
                update: 'do?sc=xworker.app.view.extjs.server.DataObjectStore/@update',
                destroy: 'do?sc=xworker.app.view.extjs.server.DataObjectStore/@destroy'
            },
            listeners: {
                load: function(proxy, o, options){
                    var grid = Ext.getCmp('gridPanel');
                    
                    var reader = grid.store.reader;
                    if(reader.jsonData){
                        if(reader.jsonData.columnModel){
                            var grid = Ext.getCmp('gridPanel');
                            var colModel = new Ext.grid.ColumnModel(reader.jsonData.columnModel);
                            
                            //判断新的表头列是否和原有列相同
                            var same = colModel.config.length == grid.colModel.config.length;
                            if(same){
                                for(i=0; i<colModel.config.length; i++){
                                    if(colModel.config[i].id != grid.colModel.config[i].id){
                                        same = false;
                                        break;
                                    }
                                }
                            }
                            
                            if(!same){
                                //不相同，重新初始化表格
                                var store = grid.store;
                                
                                grid.reconfigure(store, colModel);
                                Ext.apply(store.baseParams, store.lastOptions.params);
                            }
                        }
                    }
                }
            }
        }),
        writer: new Ext.data.JsonWriter({
            encodeDelete: true
        }),
        fields: [
            {
                mapping: 'id',
                name: 'id',
                type: 'int'
            },
            {
                mapping: 'value',
                name: 'value'
            }
        ],
        baseParams: {
            dataObjectPath:'xworker.example.app.dataobject.ProvinceDataObject'
        },
        listeners: {
            exception: function(proxy, type, action, options, response, arg){
                
                var msg = '';
                if(Ext.isDefined(response.responseText)){
                    var o = Ext.util.JSON.decode(response.responseText);
                    msg = o.msg;
                }else{
                    msg = response.message;
                }
                if(msg){
                    Ext.Msg.alert(type, msg);
                }else{
                    Ext.Msg.alert(type, '超时或发生其他错误,responseText=' + response.responseText);
                }
            }
        }
    });

    var viewport = new Ext.Viewport({
        forceLayout: true,
        layout: 'fit',
        id: 'viewport',
        renderTo: document.body,
        items: [
            new Ext.grid.EditorGridPanel({
                title: '表格',
                id: 'gridPanel',
                store: gridPanel_store,
                buttons: [
                    new Ext.Button({
                        text: '添加',
                        handler: function(){
                            var gridStore = Ext.getCmp('gridPanel').store;
                            var gridStoreRecord = {};
                            if(Ext.isArray(gridStoreRecord)){
                                var gridStoreRecordS = [];
                                for(i=0; i<gridStoreRecord.length; i++){
                                    gridStoreRecordS[i] = new gridStore.recordType(gridStoreRecord[i]);
                                }
                                gridStore.insert(gridStore.getCount(), gridStoreRecordS);
                            }else{
                                gridStore.insert(gridStore.getCount(), new gridStore.recordType(gridStoreRecord));
                            }
                        }
                    }),
                    new Ext.Button({
                        text: '删除',
                        handler: function(){
                            var xworker = Ext.ns('Ext.xworker');    
                            var grid = Ext.getCmp('gridPanel');
                            xworker.util.removeGridSelectedRows(grid, '删除记录', '确定要删除指定的记录么？');
                        }
                    }),
                    new Ext.Button({
                        text: '刷新',
                        handler: function(){
                            var store = Ext.getCmp('gridPanel').store;
                            store.reload();
                        }
                    })
                ],
                columns: [
                    {
                        dataIndex: 'id',
                        header: '标识',
                        id: 'id',
                        editor: new Ext.form.TextField({
                            name: 'id'
                        })
                    },
                    {
                        dataIndex: 'value',
                        header: '值',
                        id: 'value',
                        editor: new Ext.form.TextField({
                            name: 'value'
                        })
                    }
                ],
                loadMask: {
                    msg: '正在装载数据，请稍后...'
                }
            })
        ]
    });

});
