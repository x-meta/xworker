def charset = self.doAction("getCharset", actionContext);
if(mimeType != null && mimeType != "" && charset != null && charset != ""){
    contentType = ContentType.create(mimeType, charset);
} else if(mimeType != null){
    contentType = ContentType.create(mimeType);
}

def bytes = self.doAction("getBytes", actionContext);
//def bytes = Ognl.getValue(self.bytes, actionContext);
if(bytes == null){
    return null;
}else{
    if(contentType != null){
        return new ByteArrayEntity(bytes, contentType);
    }else{
        return new ByteArrayEntity(bytes);
    }
}
