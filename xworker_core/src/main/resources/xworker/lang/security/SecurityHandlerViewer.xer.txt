import xworker.security.SecurityManager;
import xworker.util.XWorkerUtils;
import xworker.swt.util.XWorkerTreeUtil;

import org.eclipse.swt.SWT;
import org.eclipse.swt.widgets.TreeItem;

//先清空
for(item in handlerTree.getItems()){
    item.dispose();
}

def types = ["URL", "HTTPACTION", "APP"];
for(int i=0; i<types.size(); i++){
    def type = types.get(i);
    def item = new TreeItem(handlerTree, SWT.NONE);
    item.setText(type);
    item.setData("URL");
    if(i == 0){
        item.setImage(urlImage);
    }else if(i == 1){
        item.setImage(acImage);
    }else if(i == 2){
        item.setImage(appImage);
    }
    
    def handlers = 	SecurityManager.getHandlers((byte) (i +1));
    def things = [];
    for(handler in handlers){
        things.add(handler.getThing());
    }
    
    XWorkerTreeUtil.showThingsOnTreeByGroup(things, item, actionContext);
}
