<p>JDBC数据源，对应javax.sql.DataSource，可以配置连接池等。</p>

<p><strong>主要方法：</strong></p>

<ul>
	<li><strong>getDataSource</strong><br />
	返回javax.sql.DataSource对象。<br />
	&nbsp;</li>
	<li><strong>getConnection</strong><br />
	返回java.sql.Connection连接。</li>
</ul>

<p><strong>数据源配置使用的优先级</strong></p>

<ol>
	<li><strong>refDataSource</strong><br />
	如果不为空那么认为是引用的其他数据源。</li>
	<li><strong>dbDataSourceName</strong><br />
	如果不为空那么从数据库中读取相应的配置。</li>
	<li><strong>配置文件</strong><br />
	如果不为空从配置文件中读取配置。</li>
	<li><strong>本事物</strong><br />
	最后使用的是本事物的配置。</li>
</ol>

import org.eclipse.swt.SWT;

import org.eclipse.swt.widgets.MessageBox;

import org.xmeta.ActionContext;

def coolBar = actionContext.get("coolBar");
def thing = actionContext.get("currentThing");
def log = actionContext.get("log");
def display = coolBar.getShell().display;
new Thread({
    display.asyncExec({
        try{
            def ac = new ActionContext();
            def con = thing.doAction("getConnection", actionContext);
            if(con == null){
                throw new Exception("返回的连接为null！");
            }
            con.close();
            
            def box = new MessageBox(coolBar.getShell(), SWT.OK | SWT.ICON_INFORMATION);
            box.setText("提示");
            box.setMessage("已获取连接，测试成功！");
            box.open();
        }catch(Exception e){
            log.error("测试获取连接失败", e);
            def box = new MessageBox(coolBar.getShell(), SWT.OK | SWT.ICON_INFORMATION);
            box.setText("提示");
            box.setMessage("测试获取连接失败，" + e);
            box.open();
        }
    });
}).start();
