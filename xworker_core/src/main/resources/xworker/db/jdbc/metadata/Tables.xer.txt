import xworker.dataObject.DataObject;

def catalog = actionContext.get("catalog");
def schemaPattern = actionContext.get("schemaPattern");
def tableNamePattern = actionContext.get("tableNamePattern");
def types = actionContext.get("types");

def datas = [];
def metadata = con.getMetaData();
def rs = metadata.getTables(catalog, schemaPattern, tableNamePattern, types);
try{    
    def thing = world.getThing("xworker.db.jdbc.metadata.Tables");
    while(rs.next()){
        def dataObject = new DataObject("xworker.db.jdbc.metadata.Tables");
        dataObject.setInited(false);
        for(attr in thing.get("attribute@")){
            dataObject.put(attr.name, rs.getString(attr.fieldName));
        }
        dataObject.setInited(true);
        datas.add(dataObject);
    }
}finally{
    rs.close();
}
return datas;
