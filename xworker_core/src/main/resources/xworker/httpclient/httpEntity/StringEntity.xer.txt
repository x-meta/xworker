def charset = self.doAction("getCharset", actionContext);//UtilString.getString(self, "charset", actionContext);
def content = self.doAction("getContent", actionContext);//UtilString.getString(self, "content", actionContext);

if(content == null){
    log.warn("StringEntity: content is null, thing=" + self.getMetadata().getPath());
    return null;
}

if(mimeType == null && charset == null ){
    return new StringEntity(content);
}else if(charset != null){
    return new StringEntity(content, charset);
}else {
    return new StringEntity(content, mimeType, charset);
}
