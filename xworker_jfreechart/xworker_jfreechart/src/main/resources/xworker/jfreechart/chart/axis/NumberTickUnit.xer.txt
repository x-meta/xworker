import org.jfree.chart.axis.NumberTickUnit;

import java.text.DecimalFormat;

import ognl.Ognl;

if(self.type == "Var"){
    return Ognl.getValue(self.varName, actionContext);
}

if(self.formatter == null || self.formatter == ""){
    return new NumberTickUnit(self.getDouble("size"));
}else if(self.minorTickCount == null || self.minorTickCount == ""){
    return new NumberTickUnit(self.getDouble("size"), new DecimalFormat(self.formatter))
}else{
    return new NumberTickUnit(self.getDouble("size"), new DecimalFormat(self.formatter), self.getInt("minorTickCount"))
}
