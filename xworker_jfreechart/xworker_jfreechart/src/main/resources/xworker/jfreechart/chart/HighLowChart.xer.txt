import ognl.Ognl;

import org.jfree.chart.ChartFactory;

//创建数据集
def dataset = self.doAction("getDataset", actionContext);

//创建图表
def chart = ChartFactory.createHighLowChart(self.title, self.timeAxisLabel, 
        self.valueAxisLabel, dataset,self.getBoolean("legend"));

//初始化JFreechart的属性
for(child in self.getChilds()){
    if(child.thingName != "Dataset"){
        child.doAction("init", actionContext);
    }
}

return chart;
