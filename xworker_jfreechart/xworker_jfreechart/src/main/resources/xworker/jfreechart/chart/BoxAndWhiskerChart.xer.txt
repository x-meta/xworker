import ognl.Ognl;

import org.jfree.chart.ChartFactory;

//创建数据集
def dataset = self.doAction("getDataset", actionContext);

//创建图表
def chart = ChartFactory.createBoxAndWhiskerChart(self.title, self.axisLabel, self.valueAxisLabel, dataset, self.getBoolean("legend"));

//初始化JFreechart的属性
self.doAction("initChart", actionContext, ["jfreeChart": chart]);

//初始化图表
try{
    def bindings = actionContext.push(null);
    bindings.put("chart", chart);
    self.doAction("init", actionContext);
    self.doAction("initCategoryPlot", actionContext);
    
    bindings.put("renderer", chart.getPlot().getRenderer());
    for(renderer in self.get("Renderer@")){
        renderer.doAction("init", actionContext);
    }
}finally{
    actionContext.pop();
}

return chart;
