Scode
#$@text#$@
import ognl.Ognl;

import org.jfree.chart.renderer.category.StandardBarPainter;
import org.jfree.chart.renderer.category.GradientBarPainter;

switch(self.type){
    case "Var":
        return Ognl.getValue(self.varName, actionContext);
    case "StandardBarPainter":
        return new StandardBarPainter();
    case "GradientBarPainter":
        if(self.g1 != null && self.g1 != ""){
            return new GradientBarPainter(self.getDouble("g1"), self.getDouble("g2"), self.getDouble("g3"));
        }else{
            return new GradientBarPainter();
        }
}

return null;
