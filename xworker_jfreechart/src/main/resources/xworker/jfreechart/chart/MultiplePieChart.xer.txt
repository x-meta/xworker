import ognl.Ognl;

import org.jfree.chart.ChartFactory;
import org.jfree.chart.plot.PlotOrientation;
import org.jfree.util.TableOrder;

//创建数据集
def dataset = self.doAction("getDataset", actionContext);

//创建图表
def order = TableOrder.BY_ROW;
if(self.value == "BY_COLUMN "){
    return TableOrder.BY_COLUMN ;
}

def chart = ChartFactory.createMultiplePieChart(self.title, dataset, order, self.getBoolean("legend"), 
        self.getBoolean("tooltips"), self.getBoolean("urls"));

//初始化JFreechart的属性
for(child in self.getChilds()){
    if(child.thingName != "Dataset"){
        child.doAction("init", actionContext);
    }
}

return chart;
