<p>雷达图、极坐标图等。</p>
<p>X坐标值是角度。</p>
<div class="block">Creates a polar plot for the specified dataset (x-values interpreted as angles in degrees). The chart object returned by this method uses a <a title="class in org.jfree.chart.plot" href="http://www.jfree.org/jfreechart/api/javadoc/org/jfree/chart/plot/PolarPlot.html"><code>PolarPlot</code></a> instance as the plot, with a <a title="class in org.jfree.chart.axis" href="http://www.jfree.org/jfreechart/api/javadoc/org/jfree/chart/axis/NumberAxis.html"><code>NumberAxis</code></a> for the radial axis.</div>

#$@text#$@
import ognl.Ognl;

import org.jfree.chart.ChartFactory;

//创建数据集
def dataset = self.doAction("getDataset", actionContext);

//创建图表
def chart = ChartFactory.createPolarChart(self.title, dataset, self.getBoolean("legend"), 
        self.getBoolean("tooltips"), self.getBoolean("urls"));

//初始化JFreechart的属性
for(child in self.getChilds()){
    if(child.thingName != "Dataset"){
        child.doAction("init", actionContext);
    }
}

return chart;
