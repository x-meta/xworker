import ognl.Ognl;

import org.jfree.chart.ChartFactory;

//创建数据集
def dataset = self.doAction("getDataset", actionContext);

//创建图表
def chart = ChartFactory.createGanttChart(self.title, self.categoryAxisLabel, 
        self.dateAxisLabel, dataset, self.getBoolean("legend"), 
        self.getBoolean("tooltips"), self.getBoolean("urls"));

//初始化JFreechart的属性
for(child in self.getChilds()){
    if(child.thingName != "Dataset"){
        child.doAction("init", actionContext);
    }
}

return chart;
