import ognl.Ognl;
import org.jfree.chart.labels.StandardCategoryItemLabelGenerator;
import org.jfree.chart.labels.IntervalCategoryItemLabelGenerator;
import java.text.SimpleDateFormat;
import java.text.DecimalFormat;

switch(self.type){
    case "Var":
        return Ognl.getValue(self.varName, actionContext);
    case "IntervalCategoryItemLabelGenerator":
        if(self.idateFormat != null && self.idateFormat != ""){
            return new IntervalCategoryItemLabelGenerator(self.ilabelFormat, new SimpleDateFormat(self.idateFormat));
        }else{
            return new IntervalCategoryItemLabelGenerator(self.ilabelFormat, new DecimalFormat(self.inumberFormat));
        }
    case "StandardCategoryItemLabelGenerator":    
       if(self.sdateFormat != null && self.sdateFormat != ""){
           return new StandardCategoryItemLabelGenerator(self.slabelFormat, new SimpleDateFormat(self.sdateFormat));
       }else{
           return new StandardCategoryItemLabelGenerator(self.slabelFormat, new DecimalFormat(self.snumberFormat), new DecimalFormat(self.percentFormatter));
       }
}

return null;
