#$@text#$@
import org.jfree.data.Range;
import org.jfree.data.time.DateRange;

import ognl.Ognl;

import java.text.SimpleDateFormat;

switch(self.type){
    case "Var":
        return Ognl.getValue(self.varName, actionContext);
    case "double":
        return new Range(self.getDouble("lower"), self.getDouble("upper"));
    case "date":
        def sf = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
        return new DateRange(sf.parse(self.lower), sf.parse(self.upper));
}

return null;
#$@text#$@
