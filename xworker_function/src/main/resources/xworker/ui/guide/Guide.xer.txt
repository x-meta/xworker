import org.xmeta.ActionContext;

//获取事物表单事物
def thing = world.getThing(self.getString("thingPath"));
if(thing == null){
    thing = self.getThing("thing@0");
}

if(thing == null){
    guide.showMessage("Thing is null");
}else{
    //创建事物表单
    def ac = new ActionContext();
    def composite = guide.getRequestUI();
    ac.put("parent", composite);
    ac.put("guide", guide);
    ac.put("guideThing", self);
    
    def compositeThing = world.getThing("xworker.ui.guide.ThingFormGuideShell/@mainComposite");
    compositeThing.doAction("create", ac);
    
    //设置表单
    ac.thingForm.doAction("setDescriptor", ac, ["descriptor": thing]);
    
    //设置表单的值
    def values = self.doAction("getInitValues", actionContext);
    ac.thingForm.doAction("setValues", ac, ["values": values]);
    def buttonText = self.getStringBlankAsNull("buttonText");
    if(buttonText != null){
        ac.okButton.setText(buttonText);
    }
    composite.layout();
}

import org.xmeta.ActionContext;
import xworker.util.XWorkerUtils;

def type = self.getStringBlankAsNull("type");
def path = self.getString("path");
def url = null;
if(type == "webControl"){
    def control = world.getThing(path);
    url = XWorkerUtils.getWebControlUrl(control);
}else if(type == "url"){
    url = path;
}else{
    url = XWorkerUtils.getThingDescUrl(self.getMetadata().getPath());
}
if(url == null){
    guide.showMessage("Please set url");
    return;
}

//创建浏览器
def ac = new ActionContext();
def composite = guide.getRequestUI();
ac.put("parent", composite);
ac.put("guide", guide);
ac.put("guideThing", self);

def compositeThing = world.getThing("xworker.ui.guide.HtmlGuideShell/@mainComposite"); 
compositeThing.doAction("create", ac);
ac.browser.setUrl(url);
def buttonText = self.getStringBlankAsNull("buttonText");
if(buttonText != null){
    ac.okButton.setText(buttonText);
}

composite.layout();
