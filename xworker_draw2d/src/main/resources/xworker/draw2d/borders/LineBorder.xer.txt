mport org.eclipse.draw2d.LineBorder;
import org.eclipse.draw2d.Graphics;

import org.xmeta.util.UtilString;

LineBorder border = new LineBorder();

def color = UtilString.createObjectFromParams(self.color, "xworker.swt.graphics.Color", actionContext);
if(color != null){
    border.setColor(color);
}
if(self.style != null && self.style != ""){
    switch(self.style){
        case "CUSTOM":
            border.setStyle(Graphics.LINE_CUSTOM);
            break;
        case "DASH":
            border.setStyle(Graphics.LINE_DASH);
            break;
        case "DASHDOT":
            border.setStyle(Graphics.LINE_DASHDOT);
            break;
        case "DASHDOTDOT":
            border.setStyle(Graphics.LINE_DASHDOTDOT);
            break;
        case "DOT":
            border.setStyle(Graphics.LINE_DOT);
            break;
        case "SOLID":
            border.setStyle(Graphics.LINE_SOLID);
            break;
    }
}

return border;
