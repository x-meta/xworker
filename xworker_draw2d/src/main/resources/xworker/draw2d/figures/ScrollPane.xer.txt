#$@text#$@
import org.eclipse.draw2d.ScrollPane;

def figure = new ScrollPane();

if(self.horizontalScrollBarVisibility != null && self.horizontalScrollBarVisibility != ""){
    figure.setHorizontalScrollBarVisibility(getVisibility(self.horizontalScrollBarVisibility));
}
if(self.scrollBarVisibility != null && self.scrollBarVisibility != ""){
    figure.setScrollBarVisibility(getVisibility(self.scrollBarVisibility));
}
if(self.verticalScrollBarVisibility != null && self.verticalScrollBarVisibility != ""){
    figure.setVerticalScrollBarVisibility(getVisibility(self.verticalScrollBarVisibility));
}

return figure;

def getVisibility(value){
    switch(value){
        case "ALWAYS":
            return ScrollPane.ALWAYS;
        case "AUTOMATIC":
            return ScrollPane.AUTOMATIC;
        case "NEVER":
            return ScrollPane.NEVER;  
    }
    return ScrollPane.AUTOMATIC;
}
