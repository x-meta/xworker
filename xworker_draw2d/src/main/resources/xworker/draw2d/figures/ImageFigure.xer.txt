import org.eclipse.draw2d.ImageFigure;
import org.eclipse.draw2d.PositionConstants;

import org.xmeta.util.UtilString;

def figure = new ImageFigure();

def image = UtilString.createObjectFromParams(self.image, "xworker.swt.graphics.Image", actionContext);
if(image != null){
    figure.setImage(image);
}
switch(self.alignment){
    case "NORTH":
        figure.setAlignment(PositionConstants.NORTH);
        break;
    case "SOUTH":
        figure.setAlignment(PositionConstants.SOUTH);
        break;
    case "EAST":
        figure.setAlignment(PositionConstants.EAST);
        break;
    case "WEST":
        figure.setAlignment(PositionConstants.WEST);
        break;
    case "CENTER":
        figure.setAlignment(PositionConstants.CENTER);
        break;
}

return figure;
