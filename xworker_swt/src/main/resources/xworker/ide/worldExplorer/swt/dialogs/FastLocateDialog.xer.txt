import org.eclipse.swt.SWT;
import org.eclipse.swt.widgets.TreeItem;

//查找
String searchFor = searchText.getText();

def searchThing = currentThing;
if(currentButton.getSelection()){
    searchThing = currentThing;
}else{
    searchThing = thing;
}

def results = [];
if(nodeButton.getSelection()){
    searchNode(searchFor.toUpperCase(), searchThing, results);
} else if(attributeNameButton.getSelection()){
    serachName(searchFor.toUpperCase(), searchThing, results);
} else {
    searchValue(searchFor.toUpperCase(), searchThing, results);
}   

//更新结果树
for(item in resultTree.getItems()){
    item.dispose();
}
for(result in results){
    def treeItem = new TreeItem(resultTree, SWT.NONE);
    treeItem.setText([result.name] as String[]);
    
    for(child in result.childs){
        def subItem  = new TreeItem(treeItem, SWT.NONE);
        subItem.setText([child.metadata.name, child.metadata.label, child.getThingName(),  getPath(child)] as String[]);
        subItem.setData(child);
    }
    
    treeItem.setExpanded(true);
}

//添加一个搜索结果到结果集中
def addResult(thing, results){
    def thingName = thing.getThingName();
    def rset = null;
    for(r in results){
        if(r.name == thingName){
            rset = r;
            break;
        }
    }
    
    if(rset == null){
        rset = ["name":thingName, childs:[]];
        addToList(rset, results);
    }
    addToList(thing, rset.childs);
}

//按照名字的先后顺序加入数据
def addToList(data, list){
    def added = false;
    for(int i=0; i<list.size(); i++){
        if(data.name < list.get(i).name){
            list.add(i, data);
            added = true;
            break;
        }
    }
    
    if(!added){
        list.add(data);
    }
}
def getPath(dataObj){
    String path = dataObj.metadata.name.toUpperCase();
    if(dataObj.getParent() != null){
        path = dataObj.getParent().metadata.name + "/" + dataObj.metadata.name;
    }
    
    return path;
}

def searchNode(str, dataObj, results){
    for(objName in dataObj.getThingNames()){
        //println objName;
        if(objName.toUpperCase().indexOf(str) != -1){        
            addResult(dataObj, results);
            break;    
        }
    }
    
    for(childObj in dataObj.childs){
        searchNode(str, childObj, results);
    }
}

def serachName(str, dataObj, results){
    def map = dataObj.getAttributes();
    for(key in map.keySet()){
        if(key.toUpperCase().indexOf(str) != -1){            
            addResult(dataObj, results);
        }
    }
    
    for(childObj in dataObj.childs){
        serachName(str, childObj, results);
    }
}

def searchValue(str, dataObj, results){
    def map = dataObj.getAttributes();
    for(key in map.keySet()){
        def v = map.get(key);
        if(v instanceof String && v.toUpperCase().indexOf(str) != -1){
            addResult(dataObj, results);
            break;
        }
    }
    
    for(childObj in dataObj.childs){
        searchValue(str, childObj, results);
    }
}

import org.eclipse.swt.SWT;

//打开编辑器相应的节点
def dataObj = resultTree.getSelection()[0].getData();
if(dataObj == null){
    return;
}

def it = null;
for(item in parentContext.innerOutline.getItems()){
    it = getItem(dataObj, item);
    if(it != null) break;
}

it.getParent().setSelection(it);	
it.getParent().notifyListeners(SWT.Selection, null);

def getItem(dataObj, item){
    if(item.getData().metadata.path == dataObj.metadata.path){
        return item;
    }
    
    for(subItem in item.getItems()){
        def it = getItem(dataObj, subItem);
        if(it != null) return it;
    }
    
    return null;
}
