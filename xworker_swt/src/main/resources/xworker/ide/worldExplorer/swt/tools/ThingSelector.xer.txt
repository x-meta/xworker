            currentItem = selectItem(dataTree, catPath, currentItem, treeSelection);
            //log.info("currentItem=" + currentItem);
            if(currentItem == null){
                 break;
            }else{
                catPath = catPath + ".";
            }
        }
    }
}

//子事物
if(currentItem != null && paths.length >=4 ){
    def category = paths[3];
    def cats = category.split("[/]");
    def catPath = paths[0] + ":" + paths[1] + ":" + paths[2] + ":/";
    //log.info("category=" + category);
    for(cat in cats){
        if(cat != ""){
            catPath = catPath + cat;
            //log.info("catPath=" + catPath);
            currentItem = selectItem(dataTree, catPath, currentItem, treeSelection);
            //log.info("currentItem=" + currentItem);
            if(currentItem == null){
                 break;
            }else{
                catPath = catPath + "/";
            }
        }
    }
}

def selectItem(dataTree, path, treeItem, treeSelection){
    for(item in treeItem.items){   
        def dpath = item.getData().path;
        //if(item.getData().getPath() == path){
        if(dpath == path){
            dataTree.select(item);
            treeSelection.handleEvent(null);
            return item;
        }
    }
}
