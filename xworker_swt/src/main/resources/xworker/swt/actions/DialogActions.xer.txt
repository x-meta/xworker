def ac = new ActionContext();
def parentShell = self.doAction("getShell", actionContext);

ac.put("dialogThing", self);

def dialogThing = world.getThing("xworker.swt.actions.prototypes.ThingFormDialog");
def shell = dialogThing.doAction("create", ac);

//事物表单
def thingForm = ac.thingForm;
//要编辑的事物
def thing = self.doAction("getThing", actionContext);
//描述者
def descriptor = self.doAction("getDescriptor", actionContext);
if(descriptor == null && thing != null){
    descriptor = thing.getDescriptor();
}

//设置表单和值
if(descriptor != null){
    thingForm.doAction("setDescriptor", ac, "descriptor", descriptor);
}
if(thing != null){
    thingForm.doAction("setValues", ac, "values", thing.getAttributes());
}

//打开表单
def title = self.doAction("getTitle", actionContext);
if(title != null){
    shell.setText(title);
}

shell.pack();

SwtUtils.centerShell(shell);
shell.setVisible(true);

import org.xmeta.ActionContext;

//创建新的变量上下文
def ac = new ActionContext();
def parentShell = self.doAction("getShell", actionContext);
//println parentShell;
ac.put("parent", parentShell);
ac.put("parentContext", actionContext);
ac.put("dialogThing", self);

//创建对话框
def dialogThing = world.getThing("xworker.swt.actions.prototypes.ThingEditorDialog");
def shell = dialogThing.doAction("create", ac);

//事物表单
def thingEditor = ac.thingEditor;
//要编辑的事物
def thing = self.doAction("getThing", actionContext);

//设置表单和值
if(thing != null){
    thingEditor.doAction("setThing", ac, "thing", thing);
}

//打开表单
def title = self.doAction("getTitle", actionContext);
if(title != null){
    shell.setText(title);
}

shell.pack();
shell.setVisible(true);

    ac.stackTraceText.doAction("setThrowable", ac, "throwable", throwable);
