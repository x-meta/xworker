def isMatchThing(String key, Thing thing){
    if(isMatch(key, thing.getStringBlankAsNull("th_keywords"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("group"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("en_group"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("label"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("en_label"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("zh_label"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("zh_group"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("name"))){
        return true;
    }
    
    return false;
}

def isMatch(String key, String value){
    if(value == null || "" == value){
        return false;
    }else{
        return value.toLowerCase().indexOf(key) != -1
    }
}

def initDynamicNode(item, thing, actionContext){
    synchronized(item){
        def key = "__dynamic_node_inited__";
        if(item.getData(key) == null){
            //还没有初始化
            item.setData(key, true);
            def display = item.getParent().display;
            new Thread({
                def childs = thing.doAction("getChilds", actionContext);
                display.asyncExec({
                    if(childs != null){
                        try{
                            for(child in childs){
                                def childItem = new TreeItem(item, SWT.NONE);
                                childItem.setData(child);
                                XWorkerTreeUtil.initItem(childItem, child, actionContext)
                            }
                        }catch(Exception e){
                            e.printStackTrace();
                        }
                    }
                });
            }).start();
        }
    }
}

def getViewer(thing){
    //首先取自己的
    //println thing;
    def viewer = getViewer_(thing, thing.getStringBlankAsNull("th_thingRegistViewer"));
    
    if(viewer == null){
        for(desc in thing.getAllDescriptors()){
            //println desc;
            viewer = getViewer_(thing, desc.getStringBlankAsNull("th_thingRegistViewer"));
            if(viewer != null){
                return viewer;
            }
        }
    }

    if(viewer == null){
        for(extd in thing.getAllExtends()){
            // println "exted:" extd;
            viewer = getViewer_(thing, extd.getStringBlankAsNull("th_thingRegistViewer"));
            if(viewer != null){
                return viewer;
            }
        }
    }
    
    return viewer;
}

def getViewer_(thing, path){
    if(path == null){
        return null;
    }else if(path == "self"){
        return thing;
    }else{
        return world.getThing(path);
    }
}

    if(upButton.isEnabled()){
         upButtonSelection.handleEvent(null);
         return;
    }

    }

    }

    if(nextItem != null){
         thingTree.select(nextItem);
         thingTreeSelection.handleEvent(null);
    }

    //左
    if(item.getParentItem() != null){
         thingTree.select(item.getParentItem());
         thingTreeSelection.handleEvent(null);
    }

    //回车
    thingTreeDefaultSelection.handleEvent(null);

    thingTree.showSelection();
}

def getNextItem(thingTree, item){
    if(item == null){
        return null;
    }
    
    def parent = item.getParentItem();
    if(parent == null){
        parent = thingTree;
    }
    
    def next = false;
    for(child in parent.getItems()){
        if(next){
            return child;
        }else if(child == item){
            next = true;
        }
    }
    
    return getNextItem(thingTree, item.getParentItem());
}

def getPreItem(thingTree, item){
    if(item == null){
        return null;
    }
    
    def parent = item.getParentItem();
    if(parent == null){
        parent = thingTree;
    }
    
    def preItem = null;
    for(child in parent.getItems()){
        if(child == item){
            if(preItem == null){
                return item.getParentItem();
            }else{
                return preItem;
            }
        }else{
            preItem = child;
        }
    }
    
    return getPreItem(thingTree, item.getParentItem());
}

    def index = gs[i].indexOf("|");
    if(index >= 0){
        gs[i] = gs[i].substring(index + 1, gs[i].length());
    }
}

def findItem(item, groups, index){
    if(index >= groups.length){
        return;
    }
    
    def group = groups[index];
    if(item.getText() == group){
        thingTree.setSelection(item);
    }
    for(citem in item.getItems()){
        findItem(citem, groups, index + 1);
    }
}

def have(thing, things){
   for(t in things){
       if(thing == t){
           return true;
       }else if(thing.getMetadata().getPath() == t){
           return true;
       }
   }
   
   return false;
}

def setChecked(item, things){
    def data = item.getData();
    if(data instanceof Thing){
        if(have(data, things)){
            item.setChecked(true);
        }else{
            item.setChecked(false);
        }
    }else{
        item.setChecked(false);
    }
    
    for(childItem in item.getItems()){
        setChecked(childItem, things);
    }
}

def isMatchThing(String key, Thing thing){
    if(isMatch(key, thing.getStringBlankAsNull("th_keywords"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("group"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("en_group"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("label"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("en_label"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("zh_label"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("zh_group"))){
        return true;
    }
    if(isMatch(key, thing.getStringBlankAsNull("name"))){
        return true;
    }
    
    return false;
}

def isMatch(String key, String value){
    if(value == null || "" == value){
        return false;
    }else{
        return value.toLowerCase().indexOf(key) != -1
    }

def modifyItem(treeItem, oldGroup, newGroup, attrName, result){
    def data = treeItem.getData();
    if(data instanceof Thing){
         def thing = world.getThing(data.getMetadata().getPath());
         if(thing.getMetadata().getThingManager().isSaveable() == false){
              return;
         }
         if(thing != null){
             def group = thing.getStringBlankAsNull(attrName);
             if(group != null){
                 def changed = false;
                 def gstr = null;
                 for(g in group.split("[,]")){
                     if(g.startsWith(oldGroup)){
                          g = newGroup + g.substring(oldGroup.length(), g.length());                          
                          changed = true;
                     }
                     if(gstr != null){
                          gstr = gstr + "," + g;
                      }else{
                          gstr = g;
                      }
                 }
                 
                 if(changed){
                     thing.set(attrName, gstr);
                     thing.save();
                 }
             }
         }
    }
    
    for(item in treeItem.getItems()){
        modifyItem(item, oldGroup, newGroup, attrName, result);
    }
}
