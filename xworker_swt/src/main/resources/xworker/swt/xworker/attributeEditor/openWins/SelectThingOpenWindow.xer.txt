    //避免查找范围太大耗时，直接返回
    return [];
}

def scope = world.get(values.thingScopePath);
def datas = [];
//println values.descriptor;
if(scope instanceof Thing){
    checkThing(scope, datas, values);    
}else if(scope instanceof Category){
    for(thing in scope.iterator(true)){
        if(thing.isThing((String) values.descriptor)){
            addThing(thing, datas);
        }
    }
}

return datas;

def addThing(thing, datas){
    def data = new DataObject("xworker.swt.xworker.attributeEditor.openWins.SelectThingOpenWindow/@shell/@dataStore/@dataObjects/@ThingDataObject");
    data.put("name", thing.getMetadata().getName());
    data.put("label", thing.getMetadata().getLabel());
    data.put("path", thing.getMetadata().getPath());
    datas.add(data);
}

def checkThing(thing, datas, values){
    if(thing.isThing((String) values.descriptor)){
        addThing(thing, datas);
    }else{
        for(child in thing.getChilds()){
            checkThing(child, datas, values);
        }
    }
}
