#$@text#$@
import org.eclipse.swt.SWT;
import org.eclipse.swt.widgets.TableItem;

//--------初始化注册事物列表-----------
def textStr = text.getText();
dataList = [];
if(textStr != null && "" != textStr.trim()){
    for(t in textStr.split("[,]")){
        def ts = t.split("[|]");
        if(ts.length >= 2){
            dataList.add(["type":ts[0], "thing":ts[1]]);
        }
    }
}

for(data in dataList){
    def titem = new TableItem(registsTable, SWT.NONE);
    titem.setText([data.type, data.thing] as String[]);
    titem.setData(data);
}

//----------------初始化注册类型Combo-------
def regist = world.getThing("xworker.lang.MetaThing/@th_registThing");
types = regist.get("value@");
for(type in types){
    registTypeCombo.add(type.metadata.label);
}
registTypeCombo.select(0);
